Functions

function greet(name) {
    return `Hello, ${name}`;
}

const add = function(a, b) {
    return a + b;
};

const multiply = (a, b) => a * b;

function createUser(name, role = "user") {
    return `${name} is a ${role}`;
}

function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

let output = `Greet: ${greet("Alice")}
Add: ${add(2, 3)}
Multiply: ${multiply(2, 3)}
Create User: ${createUser("Bob")}
Sum: ${sum(1, 2, 3, 4)}`;

document.getElementById('output').textContent = output;

Explanation

  • greet(name) — Returns a greeting for the given name.
  • add(a, b) — Returns the sum of a and b.
  • multiply(a, b) — Returns the product of a and b.
  • createUser(name, role = "user") — Returns a string describing the user and their role (default is "user").
  • sum(...numbers) — Returns the sum of any number of arguments using the rest parameter and reduce.
  • The output variable uses template literals to show the results of calling these functions.
  • The result is displayed in the green output box above.
